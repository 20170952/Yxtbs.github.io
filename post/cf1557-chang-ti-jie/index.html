B.Two Tables

#include<bits/stdc++.h>
#define BetterIO ios::sync_with_stdio(false)
using namespace std;
int N,K,A[100001];
pair< int,int >B[100001];
int main(void)
{
	BetterIO;
	register int Case;cin>>Case;while(Case--)
	{
		register int i;cin>>N>>K;
		for(i=1;i<=N;i++)cin>>A[i];
		for(i=1;i<=N;i++)B[i].first=A[i],B[i].second=i;
		sort(B+1,B+N+1);
		register int Ans=0;
		for(i=2;i<=N;i++)if(B[i].second!=B[i-1].second+1)Ans++;
		if(Ans<K)puts("YES");
		else puts("NO");
	}
	return 0;
}

C.Coin Rows

Difficulty\mathtt{Difficulty}Difficulty=\mathtt{}

#include<bits/stdc++.h>
#define BetterIO ios::sync_with_stdio(false)
using namespace std;
#define int long long
#define MOD 1000000007
int N,K,Inv[200001],Fct[200001];
inline int Quickly_Power(int A,int P)
{
    register int Return=1;
    while(P)
    {
        if(P&1)Return*=A,Return%=MOD;
        A*=A,A%=MOD;
        P>>=1;
    }
    return Return%MOD;
}
signed main(void)
{
	BetterIO;
	register int i;Inv[0]=Fct[0]=1;
	for(i=1;i<=200000;i++)Fct[i]=Fct[i-1]*i%MOD,Inv[i]=Inv[i-1]*Quickly_Power(i,MOD-2)%MOD;
	register int Case;cin>>Case;while(Case--)
	{
		cin>>N>>K;
		if(N&1)
		{
			register int S=0;
			for(i=0;i<N;i+=2)S=(S+Inv[i]*Fct[N]%MOD*Inv[N-i]%MOD)%MOD;
			cout<<Quickly_Power((S+1)%MOD,K)<<endl;
		}
		else
		{
			register int S1=0,S2=0;
			for(i=0;i<N;i+=2)S1=(S1+Inv[i]*Fct[N]%MOD*Inv[N-i]%MOD)%MOD;
			for(i=0;i<=N;i++)S2=(S2+Inv[i]*Fct[N]%MOD*Inv[N-i]%MOD)%MOD;
			register int Ans1=0,Ans2=1;
			for(i=1;i<=K;i++)
			{
				Ans1=(Ans1+Ans2*Quickly_Power(S2,K-i))%MOD;
				Ans2=(Ans2*S1)%MOD;
			}
			cout<<(Ans1+Ans2)%MOD<<endl;
		}
	}
	return 0;
}
